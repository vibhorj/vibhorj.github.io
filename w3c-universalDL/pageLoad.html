2. virtual page :: new page renders in SPA context ---> push additional key _clear: true together with new page context 


	dataLayer = window.dataLayer || [];
	dataLayer.push({
		"_clear": true,
		"page.pageInfo": pageObj.pageInfo coresponding top new context
		"page.category": pageObj.category coresponding top new context
		"page.attributes": pageObj.attributes coresponding top new context
	});
